
DrawUp PROC FAR

call CalcLine1Di
mov cx,50d
mov al, color
loop9:
   mov es:[di],al
   sub di,320d
loop loop9

sub line1y,50d

call CalcLine2Di
mov cx,50d
mov al,color
loop10:
   mov es:[di],al
   sub di,320d
loop loop10

sub line2y,50d

mov lastmove,0
RET
DrawUp ENDP

loop11:

loop loop11
